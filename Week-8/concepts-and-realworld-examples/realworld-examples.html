üìù Demo 1: Fetch Posts (Valid & Invalid IDs)

Try fetching posts with different IDs. Invalid IDs will demonstrate error handling.

Async/Await + Try/Catch
async function fetchPost(postId) {
    try {
        const response = await fetch(`/posts/${postId}`);
        
        if (!response.ok) {
            throw new Error(`HTTP ${response.status}: ${response.statusText}`);
        }
        
        return await response.json();
    } catch (error) {
        console.error('Error:', error.message);
        throw error;
    }
}

üë• Demo 2: Fetch Users with Network Timeout Simulation

Test timeout handling with a delayed request. You can set a custom timeout duration.

Promise.race() Pattern
async function fetchWithTimeout(url, timeout = 5000) {
    const controller = new AbortController();
    const timeoutId = setTimeout(() => controller.abort(), timeout);
    
    try {
        const response = await Promise.race([
            fetch(url, { signal: controller.signal }),
            new Promise((_, reject) => 
                setTimeout(() => reject(new Error('Timeout')), timeout)
            )
        ]);
        clearTimeout(timeoutId);
        return response;
    } catch (error) {
        clearTimeout(timeoutId);
        throw error;
    }
}

üîó Demo 3: Multiple API Calls (Promise.all)

Fetch multiple resources simultaneously. If any request fails, the entire operation fails.

Promise.all() Pattern
async function fetchMultipleResources() {
    try {
        const [posts, users, comments] = await Promise.all([
            fetch('/posts').then(r => r.json()),
            fetch('/users').then(r => r.json()),
            fetch('/comments').then(r => r.json())
        ]);
        return { posts, users, comments };
    } catch (error) {
        console.error('One or more requests failed:', error);
        throw error;
    }
}

üõ°Ô∏è Demo 4: Resilient Fetching (Promise.allSettled)

Fetch multiple resources, but don't fail if some requests fail. See partial results.

Promise.allSettled() Pattern
async function fetchResilient() {
    const results = await Promise.allSettled([
        fetch('/posts/1').then(r => r.json()),
        fetch('/posts/invalid').then(r => r.json()), // This will fail
        fetch('/users/1').then(r => r.json())
    ]);
    
    results.forEach((result, index) => {
        if (result.status === 'fulfilled') {
            console.log(`Request ${index + 1}: Success`, result.value);
        } else {
            console.error(`Request ${index + 1}: Failed`, result.reason);
        }
    });
}
